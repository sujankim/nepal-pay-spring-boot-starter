Configure

Add to your application.yml:

nepalpay:
  khalti:
    secret-key: ${KHALTI_SECRET_KEY}
    return-url:  ${KHALTI_RETURN_URL}
    website-url: ${YOUR_WEBSITE_URL}
    sandbox: true          # false for production

  esewa:
    secret-key:   ${ESEWA_SECRET_KEY}
    product-code: ${ESEWA_PRODUCT_CODE}
    success-url:  ${ESEWA_SUCCESS_URL}
    failure-url:  ${ESEWA_FAILURE_URL}
    sandbox: true          # false for production

That is all. Spring Boot auto-configures KhaltiClient and EsewaClient beans automatically when the keys are present.

Khalti Quickstart

@Service
@RequiredArgsConstructor
public class PaymentService {

    private final KhaltiClient khaltiClient; // ← auto-injected

    // Step 1: Initiate payment
    public String startKhaltiPayment(String orderId, long amountNPR) {
        var response = khaltiClient.initiatePayment(
            KhaltiInitiateRequest.builder()
                .amount(amountNPR * 100L) // NPR → paisa
                .purchaseOrderId(orderId)
                .purchaseOrderName("Your Product")
                .build()
        );
        // Store response.pidx() in your database!
        return response.paymentUrl(); // redirect user here
    }

    // Step 2: Verify after callback
    public boolean verifyKhalti(String pidx) {
        return khaltiClient
            .lookupPayment(pidx)
            .isPaymentSuccessful();
    }
}

eSewa Quickstart

@Service
@RequiredArgsConstructor
public class PaymentService {

    private final EsewaClient esewaClient; // ← auto-injected

    // Step 1: Build signed form payload
    public EsewaFormPayload startEsewaPayment(BigDecimal amountNPR) {
        String uuid = EsewaClient.generateTransactionUuid();
        // Store uuid in your database!
        return esewaClient.buildFormPayload(amountNPR, uuid);
        // Frontend POSTs this to payload.formActionUrl()
    }

    // Step 2: Verify callback
    public boolean verifyEsewa(String encodedData) {
        return esewaClient
            .verifyCallback(encodedData)
            .isPaymentSuccessful();
    }
}
